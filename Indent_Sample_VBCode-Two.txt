Sub Test_1()
         
         Dim  Grade As String
         Dim LetterGrade As String
         
         Select Case Grade
                  Case Is >= 90
                  LetterGrade = "A"
                  Case Is >= 80
                  LetterGrade = "B"
                  Case Is >= 70
                  LetterGrade = "C"
                  Case Is >= 60
                  LetterGrade = "D"
                  Case Else
                  LetterGrade = "Sorry"
         End Select
         
End Sub
